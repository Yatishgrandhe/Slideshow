# Real project structure
class TodoList:
    def __init__(self):
        self.tasks = []
    
    def add_task(self, task):
        self.tasks.append(task)
    
    def complete_task(self, index):
        if 0 <= index < len(self.tasks):
            return self.tasks.pop(index)
        return None

# Error handling
try:
    result = 10 / 0
except ZeroDivisionError:
    print("Cannot divide by zero")

# List comprehension
squares = [x**2 for x in range(10)]

# Generator function
def fibonacci():
    a, b = 0, 1
    while True:
        yield a
        a, b = b, a + b
